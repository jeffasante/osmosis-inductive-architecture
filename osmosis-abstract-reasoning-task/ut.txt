"""
osmosis/agent.py

Full osmosis agent loop with info-gain exploration.

This is the top-level orchestrator. It ties together:
  encoder  → compress observations into latent z
  search   → find candidate programs using z
  verifier → score candidates with partial credit
  bridge   → select best program, compute gradient signal
  dsl      → execute the verified rule to act efficiently

The ARC-AGI-3 efficiency metric:
  Score = f(steps_to_verified_rule)
  Fewer exploration steps = higher score.
  This means the agent must form the right hypothesis quickly —
  not just eventually succeed.

Info-gain exploration:
  When no rule is verified yet, the agent doesn't act randomly.
  It chooses the action that maximally discriminates between
  the top candidate programs. This is active learning:
  "which observation would most update my posterior?"

  info_gain(a) = entropy reduction in program distribution after
                 observing result of action a

Episode structure:
  1. Start with empty obs_seq
  2. Encode → z (prior N(0,I) at step 0)
  3. Search → candidate programs
  4. Verify best candidate
  5. If verified: act greedily from rule, return
  6. If not: pick info-gain action, observe, append to obs_seq, repeat
"""

from __future__ import annotations

import numpy as np
from dataclasses import dataclass, field
from typing import Protocol, runtime_checkable

from osmosis.encoder  import ObsEncoder, Transition
from osmosis.dsl      import DSL, GridState, Action, Program
from osmosis.search   import search, SearchResult
from osmosis.verifier import Verifier, VerifyResult
from osmosis.bridge   import Bridge, BridgeMethod



# Environment protocol


@runtime_checkable
class ArcEnv(Protocol):
    """
    Minimal interface that an ARC-AGI-3 environment must implement.
    Concrete environments (from the benchmark) plug in here.
    """

    def reset(self) -> GridState: ...
    def step(self, action: Action) -> tuple[GridState, bool]: ...
    def current_state(self) -> GridState: ...
    def action_space(self) -> list[Action]: ...
    def is_solved(self) -> bool: ...



# Episode result


@dataclass
class EpisodeResult:
    solved:          bool
    steps:           int
    verified_rule:   Program | None
    verify_result:   VerifyResult | None
    obs_seq:         list[Transition]
    bridge_signals:  list[float] = field(default_factory=list)

    @property
    def efficiency(self) -> float:
        """
        Efficiency score in [0, 1].
        1.0 = solved in 1 step. 0.0 = budget exhausted without solving.
        Mirrors the ARC-AGI-3 scoring: penalises exploration steps.
        """
        if not self.solved:
            return 0.0
        return 1.0 / max(1, self.steps)

    def summary(self) -> str:
        status = "SOLVED" if self.solved else "FAILED"
        rule = DSL.program_to_str(self.verified_rule) if self.verified_rule else "none"
        return (f"{status} | steps={self.steps} | efficiency={self.efficiency:.3f}"
                f" | rule={rule}")



# Info-gain action selection


def info_gain_action(
    candidates:  list[Program],
    actions:     list[Action],
    state:       GridState,
    dsl:         DSL,
    n_sample:    int = 8,
) -> Action:
    """
    Choose the action that maximally discriminates between top candidates.

    For each candidate action a, simulate what each candidate program
    would predict. Actions that produce diverse predictions across
    programs are most informative — they will eliminate the most
    candidates after we observe the real outcome.

    info_gain(a) ≈ entropy of { program.predict(state, a) : program in candidates }

    High entropy means programs disagree about what would happen →
    observing this action gives us the most information.

    Parameters
    ----------
    candidates : top candidate programs from search
    actions    : available actions in current state
    state      : current environment state
    dsl        : DSL for execution
    n_sample   : max actions to evaluate (cap for speed)
    """
    if not actions:
        return Action(kind="keyboard", key="ArrowRight")

    sampled_actions = actions[:n_sample]
    best_action     = sampled_actions[0]
    best_entropy    = -1.0

    for action in sampled_actions:
        predictions = []
        for prog in candidates:
            pred = dsl.execute(prog, state, action)
            predictions.append(pred.grid)

        if len(predictions) < 2:
            continue

        # Measure disagreement: fraction of cell positions where programs disagree
        base  = predictions[0]
        diffs = [np.mean(p != base) for p in predictions[1:]]
        entropy = float(np.mean(diffs))

        if entropy > best_entropy:
            best_entropy = entropy
            best_action  = action

    return best_action



# Greedy action from verified rule


def act_from_rule(
    rule:    Program,
    state:   GridState,
    actions: list[Action],
    dsl:     DSL,
) -> Action:
    """
    Once a rule is verified, act greedily: choose the action whose
    predicted outcome best matches what the rule prescribes.

    If the rule is a pure transform (no action dependency), return
    the first available action — exploration is done.
    """
    if not actions:
        return Action(kind="keyboard", key="ArrowRight")

    # Try each action; pick the one that produces the most change
    # (heuristic: the rule is likely the "do the right thing" operator)
    best_action  = actions[0]
    best_change  = -1.0

    for action in actions:
        pred   = dsl.execute(rule, state, action)
        change = float(np.mean(pred.grid != state.grid))
        if change > best_change:
            best_change = change
            best_action = action

    return best_action



# Agent


class OsmosisAgent:
    """
    The full osmosis agent.

    Parameters
    ----------
    encoder  : trained ObsEncoder (or fresh for testing)
    dsl      : DSL instance configured for the environment
    verifier : Verifier instance
    bridge   : Bridge instance (method selection)
    budget   : max exploration steps before giving up
    top_m    : number of search candidates to maintain

    Usage
    -----
    agent = OsmosisAgent(encoder, dsl, verifier, bridge)
    result = agent.run(env)
    print(result.summary())
    """

    def __init__(
        self,
        encoder:  ObsEncoder,
        dsl:      DSL,
        verifier: Verifier,
        bridge:   Bridge,
        budget:   int = 50,
        top_m:    int = 8,
        rng_seed: int = 42,
    ):
        self.encoder  = encoder
        self.dsl      = dsl
        self.verifier = verifier
        self.bridge   = bridge
        self.budget   = budget
        self.top_m    = top_m
        self.rng      = np.random.default_rng(rng_seed)
        self._step    = 0   # training step counter (for bridge annealing)

    def run(self, env: ArcEnv) -> EpisodeResult:
        """
        Run one episode. Returns EpisodeResult.

        The loop:
          encode → search → verify → (act_from_rule | info_gain_explore)
        """
        obs_seq:        list[Transition]  = []
        bridge_signals: list[float]       = []
        state          = env.reset()

        for step in range(self.budget):

            # 1. Encode current observation history
            dist = self.encoder(obs_seq, rng=self.rng)
            z    = dist.z

            # 2. Search for candidate programs guided by z
            result: SearchResult = search(
                z        = z,
                obs_seq  = obs_seq,
                dsl      = self.dsl,
                top_m    = self.top_m,
                rng      = self.rng,
            )

            # 3. Verify the top candidate
            best_prog, vresult = self.verifier.best(result.programs, obs_seq)

            # 4. Bridge: select program, get gradient signal
            scores = np.array([
                self.verifier.verify(p, obs_seq).reward
                for p in result.programs
            ], dtype=np.float32)

            boutput = self.bridge(scores, result.programs, step=self._step)
            bridge_signals.append(boutput.gradient_signal)
            self._step += 1

            # 5. Verified? Act greedily from rule.
            if vresult.verified:
                actions = env.action_space()
                action  = act_from_rule(best_prog, state, actions, self.dsl)
                state, solved = env.step(action)

                return EpisodeResult(
                    solved        = solved or env.is_solved(),
                    steps         = step + 1,
                    verified_rule = best_prog,
                    verify_result = vresult,
                    obs_seq       = obs_seq,
                    bridge_signals = bridge_signals,
                )

            # 6. Not verified — pick info-gain action and observe
            actions = env.action_space()
            action  = info_gain_action(
                result.programs, actions, state, self.dsl
            )

            next_state, solved = env.step(action)

            # Record transition
            obs_seq.append(Transition(
                state      = state.flatten(),
                action     = action.to_vector(),
                next_state = next_state.flatten(),
            ))

            state = next_state

            if solved:
                return EpisodeResult(
                    solved        = True,
                    steps         = step + 1,
                    verified_rule = best_prog,
                    verify_result = vresult,
                    obs_seq       = obs_seq,
                    bridge_signals = bridge_signals,
                )

        # Budget exhausted
        return EpisodeResult(
            solved        = False,
            steps         = self.budget,
            verified_rule = None,
            verify_result = None,
            obs_seq       = obs_seq,
            bridge_signals = bridge_signals,
        )



# Metrics tracker (for the three measurements from the review)


@dataclass
class OsmosisMetrics:
    """
    The three metrics flagged as most important in the review:
      1. Final task success rate
      2. Exploration steps to verification
      3. Posterior concentration rate (most scientifically informative)
    """
    episodes: list[EpisodeResult] = field(default_factory=list)

    def record(self, result: EpisodeResult) -> None:
        self.episodes.append(result)

    @property
    def success_rate(self) -> float:
        if not self.episodes:
            return 0.0
        return sum(e.solved for e in self.episodes) / len(self.episodes)

    @property
    def mean_steps(self) -> float:
        solved = [e.steps for e in self.episodes if e.solved]
        return float(np.mean(solved)) if solved else float("inf")

    @property
    def mean_efficiency(self) -> float:
        if not self.episodes:
            return 0.0
        return float(np.mean([e.efficiency for e in self.episodes]))

    @property
    def posterior_concentration(self) -> float:
        """
        Measures how quickly the bridge_signals concentrate (increase)
        over the course of episodes. High concentration = encoder learns
        to assign mass to the right programs faster over time.

        Computed as: mean slope of bridge_signal over episode steps.
        Positive slope = signals improving = posterior concentrating.
        """
        slopes = []
        for e in self.episodes:
            sigs = e.bridge_signals
            if len(sigs) < 2:
                continue
            x = np.arange(len(sigs), dtype=np.float32)
            slope = float(np.polyfit(x, sigs, 1)[0])
            slopes.append(slope)
        return float(np.mean(slopes)) if slopes else 0.0

    def report(self) -> str:
        return (
            f"episodes={len(self.episodes)} | "
            f"success={self.success_rate:.2%} | "
            f"mean_steps={self.mean_steps:.1f} | "
            f"efficiency={self.mean_efficiency:.3f} | "
            f"posterior_conc={self.posterior_concentration:.4f}"
        )
"""
osmosis/bridge.py

Gradient flow between the continuous encoder and discrete verifier.

The problem (Crux 3):
  The encoder outputs a continuous latent z.
  The verifier scores discrete programs.
  The gap between them breaks gradient flow — you cannot backpropagate
  through "pick the best program from a list."

Three approaches, in order of preference:

  A. Gumbel-Softmax (preferred during training)
     Differentiable approximation to categorical sampling.
     Uses temperature τ that anneals from 1.0 → 0.1 during training.
     As τ → 0, the distribution becomes one-hot (fully discrete).
     As τ → ∞, it becomes uniform (fully continuous).

  B. Straight-Through Estimator (faster, less stable)
     Forward pass: hard argmax (discrete).
     Backward pass: pretend argmax was the identity function.
     Simple and fast, but the gradient is biased.

  C. REINFORCE / Policy Gradient (fallback)
     Treat program selection as a policy decision.
     Reward = verifier score. No differentiability needed.
     Works but is high variance — needs many samples to be stable.

This file implements all three so the training loop can switch between them.
In the numpy reference, "gradient" is computed as a scalar signal
(the direction to push z). In the torch version, these become autograd ops.
"""

from __future__ import annotations

import numpy as np
from dataclasses import dataclass
from enum import Enum, auto
from typing import Callable

from osmosis.dsl import Program
from osmosis.verifier import VerifyResult



# Bridge method selector


class BridgeMethod(Enum):
    GUMBEL_SOFTMAX      = auto()
    STRAIGHT_THROUGH    = auto()
    REINFORCE           = auto()



# Bridge output


@dataclass
class BridgeOutput:
    """
    Result of a bridge forward pass.

    selected_program : the program chosen for execution
    soft_weights     : soft distribution over candidates (for gradient)
    gradient_signal  : scalar: how much to update z (positive = good)
    method           : which bridge method was used
    """
    selected_program: Program
    soft_weights:     np.ndarray   # (k,) sums to 1
    gradient_signal:  float
    method:           BridgeMethod



# Gumbel-Softmax


def gumbel_softmax(
    logits: np.ndarray,
    tau:    float = 1.0,
    rng:    np.random.Generator | None = None,
) -> np.ndarray:
    """
    Sample from a categorical distribution using the Gumbel-Softmax trick.

    Adds Gumbel noise to logits then applies softmax with temperature tau.
    As tau → 0: approaches one-hot (argmax).
    As tau → inf: approaches uniform distribution.

    logits : (k,) unnormalized scores
    tau    : temperature (anneal from 1.0 to 0.1 during training)
    returns: (k,) soft weights summing to 1
    """
    rng = rng or np.random.default_rng()

    # Sample Gumbel noise: -log(-log(U)) where U ~ Uniform(0,1)
    U = rng.uniform(1e-8, 1.0, size=logits.shape).astype(np.float32)
    gumbel_noise = -np.log(-np.log(U))

    perturbed = (logits + gumbel_noise) / tau

    # Stable softmax
    shifted = perturbed - perturbed.max()
    exp     = np.exp(shifted)
    return exp / exp.sum()


def bridge_gumbel(
    scores:   np.ndarray,
    programs: list[Program],
    tau:      float = 1.0,
    rng:      np.random.Generator | None = None,
) -> BridgeOutput:
    """
    Select a program using Gumbel-Softmax.

    In the torch training loop, this is differentiable end-to-end.
    The gradient w.r.t. z flows back through the soft_weights.

    scores   : (k,) verifier scores for each candidate
    programs : list of k candidate programs
    tau      : annealing temperature
    """
    rng          = rng or np.random.default_rng()
    soft_weights = gumbel_softmax(scores, tau=tau, rng=rng)
    selected_idx = int(np.argmax(soft_weights))

    # Gradient signal: expected reward under soft distribution
    gradient_signal = float(np.dot(soft_weights, scores))

    return BridgeOutput(
        selected_program = programs[selected_idx],
        soft_weights     = soft_weights,
        gradient_signal  = gradient_signal,
        method           = BridgeMethod.GUMBEL_SOFTMAX,
    )



# Straight-Through Estimator


def bridge_straight_through(
    scores:   np.ndarray,
    programs: list[Program],
) -> BridgeOutput:
    """
    Select program via hard argmax.
    Backward pass uses the identity function (straight-through trick).

    Forward:  selected = argmax(scores)       [discrete]
    Backward: dL/dscores ≈ dL/d_selected      [pretend argmax = identity]

    This is biased but simple and often works well in practice.
    """
    selected_idx = int(np.argmax(scores))

    # One-hot soft weights (hard)
    soft_weights = np.zeros_like(scores, dtype=np.float32)
    soft_weights[selected_idx] = 1.0

    return BridgeOutput(
        selected_program = programs[selected_idx],
        soft_weights     = soft_weights,
        gradient_signal  = float(scores[selected_idx]),
        method           = BridgeMethod.STRAIGHT_THROUGH,
    )



# REINFORCE / Policy Gradient


def bridge_reinforce(
    scores:    np.ndarray,
    programs:  list[Program],
    rng:       np.random.Generator | None = None,
    n_samples: int = 4,
) -> BridgeOutput:
    """
    Sample programs proportional to scores, use rewards as gradient signal.

    This is the policy gradient (REINFORCE) approach. No differentiability
    needed — gradient is estimated from reward samples.

    High variance: needs n_samples > 1 to reduce noise. Use as fallback
    when Gumbel-Softmax training is unstable.

    scores    : (k,) verifier scores
    n_samples : number of programs to sample for variance reduction
    """
    rng = rng or np.random.default_rng()

    # Softmax policy
    shifted = scores - scores.max()
    exp     = np.exp(shifted)
    policy  = exp / exp.sum()                              # (k,)

    # Sample n_samples programs
    indices = rng.choice(len(programs), size=n_samples, p=policy)
    rewards = scores[indices]

    # Baseline: mean reward (reduces variance)
    baseline = rewards.mean()

    # Policy gradient: E[(reward - baseline) * log_prob]
    log_probs = np.log(policy[indices] + 1e-8)
    gradient_signal = float(np.mean((rewards - baseline) * log_probs))

    # Best sample as selected program
    best_idx = indices[np.argmax(rewards)]

    return BridgeOutput(
        selected_program = programs[best_idx],
        soft_weights     = policy,
        gradient_signal  = gradient_signal,
        method           = BridgeMethod.REINFORCE,
    )



# Bridge (unified interface)


class Bridge:
    """
    Unified bridge interface. Selects a program from candidates and
    returns a gradient signal for updating the encoder.

    The annealing schedule matters:
      Early training (step < warmup): τ = 1.0  (explore broadly)
      Late training  (step > warmup): τ → 0.1  (commit to best programs)

    Usage
    -----
    bridge = Bridge(method=BridgeMethod.GUMBEL_SOFTMAX, tau_start=1.0)
    output = bridge.forward(scores, programs, step=1000)
    # output.gradient_signal → use to update encoder
    # output.selected_program → pass to verifier for full evaluation
    """

    def __init__(
        self,
        method:        BridgeMethod = BridgeMethod.GUMBEL_SOFTMAX,
        tau_start:     float = 1.0,
        tau_end:       float = 0.1,
        anneal_steps:  int   = 10_000,
        rng_seed:      int   = 42,
    ):
        self.method       = method
        self.tau_start    = tau_start
        self.tau_end      = tau_end
        self.anneal_steps = anneal_steps
        self.rng          = np.random.default_rng(rng_seed)

    def tau(self, step: int) -> float:
        """
        Temperature at training step `step`.
        Linear anneal: tau_start → tau_end over anneal_steps.
        """
        progress = min(1.0, step / max(1, self.anneal_steps))
        return self.tau_start + progress * (self.tau_end - self.tau_start)

    def forward(
        self,
        scores:   np.ndarray,
        programs: list[Program],
        step:     int = 0,
    ) -> BridgeOutput:
        """
        Select a program and compute gradient signal.

        Parameters
        ----------
        scores   : (k,) soft verifier scores for each candidate
        programs : list of k candidate programs
        step     : current training step (for temperature annealing)
        """
        if len(programs) == 0:
            raise ValueError("programs list is empty")

        scores = np.array(scores, dtype=np.float32)

        if self.method == BridgeMethod.GUMBEL_SOFTMAX:
            return bridge_gumbel(
                scores, programs, tau=self.tau(step), rng=self.rng
            )
        elif self.method == BridgeMethod.STRAIGHT_THROUGH:
            return bridge_straight_through(scores, programs)
        elif self.method == BridgeMethod.REINFORCE:
            return bridge_reinforce(scores, programs, rng=self.rng)
        else:
            raise ValueError(f"Unknown bridge method: {self.method}")

    def __call__(self, scores, programs, step=0):
        return self.forward(scores, programs, step)
# dsl — the language of rules

## What it does in plain terms

The DSL (domain-specific language) defines what a "rule" is. It is a small vocabulary of grid operations that can be combined into programs. The verifier uses these programs to predict what the environment will do next.

The key design decision is the coarse-to-fine structure. Instead of searching over all possible programs from the start, osmosis first searches over short sketches, then refines only the promising ones into full programs.

---

## Why language design is Crux 1

If the DSL does not contain the right primitives, the correct rule cannot be expressed. No amount of clever inference will fix a vocabulary that is missing the necessary words.

At the same time, a DSL that is too expressive makes search impossibly slow. The goal is the smallest vocabulary that can express all the rules appearing in the target environments.

The primitives were chosen by examining the ARC-AGI-3 game data. Games with keyboard input tend to involve spatial transforms (translate, rotate, flip). Games with click input tend to involve color operations (recolor, select). The vocabulary covers both.

---

## Two grammar levels

```mermaid
graph TD
    A[latent z from encoder] --> B[Level 0: sketch enumeration]
    B --> C["4 096 short programs\n1-3 ops, coarse params"]
    C --> D[GPU execute on first 4 observations]
    D --> E[keep top 128 sketches]
    E --> F[Level 1: refinement]
    F --> G["512 concrete programs per sketch\nfull params, exact values"]
    G --> H[GPU execute on full observation sequence]
    H --> I[top-m candidates to verifier]

    style C fill:#FAEEDA,stroke:#854F0B,color:#633806
    style G fill:#E1F5EE,stroke:#0F6E56,color:#085041
```

**Level 0 sketches** are short sequences of high-level operations with coarse parameter bins. For example: "translate by roughly one step, then recolor." There are about 4 096 of these. They are cheap to execute and easy to enumerate.

**Level 1 programs** are fully instantiated: exact offsets, exact colors, exact region boundaries. Each sketch expands into 512 concrete programs by randomly sampling fine-grained parameters. The total search space is 4 096 x 512 = 2 million programs, but the coarse pruning means only 128 x 512 = 65 536 are fully evaluated.

---

## Primitive operations

| Category | Operations |
|---|---|
| Spatial | translate, rotate 90, flip horizontal, flip vertical, crop |
| Color | recolor, fill background, invert colors, threshold |
| Pattern | tile, mirror extend, gravity |
| Selection | select color, select region |
| Compositional | compose (A then B), conditional (if color present: A, else B) |

The compositional operations are important for hard levels. A rule like "if color 3 is present, pull everything downward; otherwise flip horizontally" requires CONDITIONAL wrapping two sub-programs.

---

## The interpreter

The interpreter is a pure function:

$$
\text{execute}(\text{program},\ s,\ a) \rightarrow s'
$$

It takes a program, an input state, and an action, and returns a predicted next state. It has no side effects and no memory.

This property is what makes GPU batch execution possible. The interpreter can be run on k programs simultaneously because each execution is independent of the others.

---

## State and action representation

A state is a grid of integers from 0 to 9, where each integer is an ARC color. The background is always 0.

An action is encoded as a fixed-length float vector so it can be passed through the encoder's triple embedder. Keyboard keys are hashed to one-hot positions. Click coordinates are mapped to separate positions.

---

## Coarse-to-fine reasoning

The two-level structure solves a real combinatorial problem. Consider enumerating all programs of length 3 with exact parameters. With 17 op codes and even modest parameter ranges, the space exceeds 10 billion programs. That is not searchable.

By separating "what type of operation" (Level 0) from "what exact parameters" (Level 1), osmosis reduces the Level 0 search to 4 096 candidates, which can be evaluated in milliseconds even on CPU. Only 128 of those advance to full evaluation.

$$
\text{effective search space} = \underbrace{4096}_{\text{sketches}} \times \underbrace{0.03}_{\text{keep top 3\%}} \times \underbrace{512}_{\text{refinements}} = 65\,536
$$

This is the number the GPU executor actually processes: tractable, and fast in parallel.
# encoder — compressing observations into a hypothesis

## What it does in plain terms

The encoder watches a sequence of interactions — "I was in state S, I took action A, the world became S'" — and produces a single compact vector called the latent z.

Think of z as a compressed belief about what rule governs the environment. After seeing one interaction, z is vague. After seeing five, it should be more specific. After seeing enough, it should point strongly at the correct rule.

---

## Why transitions, not states

Most encoders compress states: "what does this grid look like?" osmosis compresses transitions: "what happened when I intervened."

This distinction matters. Two grids can look similar but be governed by completely different rules. Two transition sequences governed by the same rule will produce a consistent pattern regardless of what the grids look like.

By encoding (state, action, next_state) triples, the encoder learns causal structure rather than visual structure.

---

## Architecture

```mermaid
graph LR
    A["(s, a, s') triples"] --> B[triple embedder]
    B --> C[self-attention layers]
    C --> D[attention pooling]
    D --> E[linear heads]
    E --> F["mu, logvar"]
    F --> G["z = mu + std * noise"]

    style G fill:#EEEDFE,stroke:#534AB7,color:#3C3489
```

**Triple embedder.** Concatenates [s, a, s'] into a single vector and passes it through a two-layer network to produce a token of size d_model. One token per interaction.

**Self-attention.** Lets each token look at every other token. A transition that contradicts the pattern can suppress misleading earlier transitions. A transition that reveals the rule can amplify consistent ones.

**Attention pooling.** Not all transitions are equally informative. Some steps are noise. Some steps reveal the rule. The pooling layer learns a query vector that assigns higher weight to more informative transitions.

$$
\text{weight}_i = \text{softmax}(q \cdot \text{token}_i)
$$

$$
\text{pooled} = \sum_i \text{weight}_i \cdot \text{token}_i
$$

**Linear heads.** Two separate linear projections produce mu (the centre of the latent distribution) and logvar (the log of its variance).

**Reparameterisation.** A sample z is drawn from the distribution without breaking gradient flow:

$$
z = \mu + e^{0.5 \cdot \text{logvar}} \cdot \varepsilon \quad \text{where} \quad \varepsilon \sim \mathcal{N}(0, I)
$$

---

## The ELBO loss

The encoder is trained with the Evidence Lower BOund (ELBO):

$$
\text{ELBO} = \underbrace{\text{reconstruction loss}}_{\text{does z lead to the right rule?}} + \beta \cdot \underbrace{D_{\text{KL}}(q(z|x) \,\|\, p(z))}_{\text{keep z close to the prior}}
$$

The reconstruction loss comes from the verifier: how well does the program found by searching near z explain the observed transitions?

The KL term keeps the latent space well-organised. Without it, the encoder can place different rules in completely separate parts of the space with no useful structure between them.

The beta parameter is annealed from 0 to 1 during training (beta-VAE schedule). Starting with beta=0 lets the encoder first learn to reconstruct correctly before being forced to organise the latent space.

---

## Empty sequence handling

At the first step of an episode, the observation sequence is empty. The encoder returns the prior: z sampled from N(0, I). This is intentional — before any observations, all rules are equally plausible.

---

## Behavioral alignment loss

The encoder also receives a contrastive signal from the proposer module. If two observation sequences are governed by the same underlying rule, their latents should be close. If governed by different rules, they should be far apart.

This is what shapes the latent space into a meaningful geometry over rules, rather than just a smooth but arbitrary compression.

$$
\text{loss}_{\text{contrastive}} = \begin{cases} \|z_i - z_j\|^2 & \text{if same rule} \\ \max(0,\ m - \|z_i - z_j\|)^2 & \text{if different rule} \end{cases}
$$

where m is a margin (default 2.0) that defines the minimum separation between different rules.
# proposer — aligning geometry with behaviour

## What it does in plain terms

The proposer enforces a crucial property: rules that behave the same way should be neighbours in the latent space. Rules that behave differently should be far apart.

Without this, the encoder might produce a beautifully smooth latent space that is completely useless for search — nearby points could correspond to rules with totally different behaviours.

---

## The alignment problem

The encoder compresses observations into a latent z. The search uses z to find candidate programs. But there is a hidden assumption: that points near z in latent space correspond to programs with similar behaviour.

This assumption is not automatically satisfied. A neural network will produce smooth embeddings, but "smooth" does not mean "semantically aligned."

The proposer fixes this by adding a contrastive loss during training.

---

## How contrastive learning works

```mermaid
graph LR
    A[program i] --> B[execute on probes]
    C[program j] --> B
    B --> D{same behaviour?}
    D -->|yes| E[pull z_i and z_j closer]
    D -->|no| F[push z_i and z_j apart]

    style E fill:#E1F5EE,stroke:#0F6E56,color:#085041
    style F fill:#FAECE7,stroke:#993C1D,color:#712B13
```

The key object is the **probe set**: a fixed collection of (state, action) pairs used to test what a program does. Two programs are behaviourally equivalent if they produce identical outputs on all probes.

The contrastive loss has two cases:

**Positive pair** (same behaviour):
$$
\mathcal{L} = \|z_i - z_j\|^2
$$
Minimising this pulls equivalent programs together in latent space.

**Negative pair** (different behaviour):
$$
\mathcal{L} = \max(0,\ m - \|z_i - z_j\|)^2
$$
Minimising this pushes non-equivalent programs apart, but only until they are separated by the margin m. Once they are far enough away, the loss is zero.

The margin m (default 2.0) prevents the encoder from collapsing everything into a single point just to minimise the positive loss.

---

## Why probes, not observed transitions

Probes are synthetic test inputs, not part of the agent's history. They are fixed during training.

Using observed transitions would create circular reasoning: the encoder would learn that programs consistent with what it has already seen are similar, which is just restating the search problem.

Probes test generalisation: "do these two programs agree on inputs they have never seen?" This is closer to the real notion of rule equivalence.

---

## Prototype embeddings

The proposer builds an index of prototype embeddings — one per behavioural equivalence class. Programs with identical behaviour share a prototype direction in latent space, with small noise added for distinctness.

When the encoder produces a z, the proposer ranks candidate programs by cosine similarity between z and their prototype:

$$
\text{score}(p) = \frac{z \cdot \text{proto}(p)}{\|z\| \cdot \|\text{proto}(p)\|}
$$

High similarity means z is pointing in the direction of that program's equivalence class.

---
---

# search — finding the right rule fast

## What it does in plain terms

The search module takes the latent z from the encoder and finds the best candidate programs. It does this in two phases to balance speed against coverage.

The key systems contribution of osmosis is the GPU-parallel execution in Phase 2. The CPU reference in `search.py` runs programs sequentially. The GPU version runs all 65 536 programs simultaneously in a single kernel call.

---

## Two-phase search

```mermaid
graph TD
    A["z from encoder"] --> B["Phase 1: coarse\n4 096 sketches\nexecute on first 4 obs"]
    B --> C["prune to top 128 by consistency score"]
    C --> D["Phase 2: fine\n128 x 512 = 65 536 programs\nexecute on full obs sequence"]
    D --> E["top-m candidates to verifier"]

    style B fill:#FAEEDA,stroke:#854F0B,color:#633806
    style D fill:#E1F5EE,stroke:#0F6E56,color:#085041
```

**Phase 1** executes 4 096 sketches against only the first 4 observations. This is fast because fewer observations means fewer interpreter calls, and sketches are short programs. The consistency score at this stage is noisy but good enough for pruning.

**Phase 2** executes 65 536 refined programs against the full observation sequence. This is the expensive step and the target for GPU acceleration.

---

## Consistency score

The fast inner-loop signal is consistency: the fraction of observed transitions that a program correctly predicts.

$$
\text{consistency}(p) = \frac{\text{correct predictions}}{|\text{obs\_seq}|}
$$

where a prediction is correct if:

$$
\text{execute}(p,\ s,\ a) = s'
$$

A program with consistency 1.0 explains every observed transition perfectly. This is the necessary condition for full verification. Consistency alone is not sufficient — the verifier adds three more signals on top of it.

---

## GPU acceleration (target)

The CPU reference loops over programs sequentially. The GPU version vectorises the interpreter across the k-batch dimension.

Concretely, instead of:

```
for program in programs:
    score = execute(program, obs_seq)
```

the GPU version runs:

```
scores = cuda_exec(programs_tensor, obs_tensor)
```

where `cuda_exec` is a custom CUDA kernel that maps each thread to one program-transition pair. This makes the execution time independent of k (the number of programs), up to hardware limits.

The CPU reference achieves roughly 28 000 programs/second on a 5x5 grid. A GPU implementation should reach 10-100 million programs/second, which is the 100-1000x speedup the architecture relies on.

---
---

# verifier — scoring how good a rule is

## What it does in plain terms

The verifier takes a candidate program and scores it on four dimensions. It returns a soft reward in [0, 1] rather than a hard pass/fail.

This softness is essential for training. Hard pass/fail means most training steps produce zero gradient because most programs are wrong. Soft scoring means the encoder always gets a signal, even when no program is fully correct yet.

---

## Four signals

```mermaid
graph LR
    P[candidate program] --> C[consistency 60%]
    P --> K[compression 15%]
    P --> AP[action pred 10%]
    P --> CF[counterfactual 15%]
    C --> R[reward]
    K --> R
    AP --> R
    CF --> R

    style R fill:#E1F5EE,stroke:#0F6E56,color:#085041
```

**Consistency (60% weight).** Fraction of observed transitions correctly predicted. This is the primary signal. A program must score 1.0 here to be marked as verified.

$$
\text{consistency} = \frac{\text{correct predictions}}{|\text{obs\_seq}|}
$$

**Compression (15% weight).** Shorter programs are preferred. ARC tasks are designed to have compact solutions. A program of length 1 scores 1.0; each additional operation reduces the score.

$$
\text{compression} = \frac{1}{\text{program length}}
$$

**Action prediction (10% weight).** A program that always returns the input state unchanged is degenerate — it has learned nothing. This signal rewards programs that produce meaningful changes.

$$
\text{action\_pred} = \frac{\text{transitions where output} \neq \text{input}}{|\text{obs\_seq}|}
$$

**Counterfactual (15% weight).** Does the program behave consistently on held-out probe states? Consistent behaviour suggests a genuine rule rather than overfitting to the observed transitions.

---

## Full verification condition

A program is fully verified only when:

1. consistency == 1.0 (explains every observed transition)
2. obs_seq is non-empty (at least one observation has been made)

The reward is soft regardless. Only the `verified` flag is hard. The agent acts greedily from a rule as soon as `verified` becomes True.

---

## Weighted reward

$$
\text{reward} = 0.60 \cdot \text{consistency} + 0.15 \cdot \text{compression} + 0.10 \cdot \text{action\_pred} + 0.15 \cdot \text{counterfactual}
$$

The weights can be adjusted. The 60% weight on consistency reflects that it is the necessary condition. The other three signals guide the encoder before any program reaches consistency 1.0.

---
---

# bridge — connecting continuous and discrete

## What it does in plain terms

The bridge solves a fundamental problem: the encoder produces a continuous latent z, but the verifier scores discrete programs. There is no obvious way to send gradients from "program 47 scored 0.8" back to "adjust z in this direction."

The bridge provides three ways to construct a differentiable path between them.

---

## Why this is hard

Consider a simple case. The encoder produces z. The search finds 10 candidate programs. The verifier scores them. You want to update z to produce better candidates next time.

But "update z" requires a gradient. And the gradient must flow through "pick a program from a list," which is a discrete, non-differentiable operation.

This is the same fundamental problem that appears in reinforcement learning, variational autoencoders with discrete latents, and neural program synthesis. osmosis inherits all three solution strategies.

---

## Three approaches

```mermaid
graph TD
    S[verifier scores] --> G[Gumbel-Softmax]
    S --> ST[Straight-Through]
    S --> RF[REINFORCE]
    G -->|preferred| Z[gradient signal to encoder]
    ST -->|faster, biased| Z
    RF -->|fallback, noisy| Z

    style G fill:#EEEDFE,stroke:#534AB7,color:#3C3489
    style Z fill:#E1F5EE,stroke:#0F6E56,color:#085041
```

**Gumbel-Softmax (preferred).** Adds noise to scores and applies a soft temperature-scaled selection. As temperature decreases, selection approaches hard argmax. As temperature increases, selection approaches uniform random.

The Gumbel noise comes from sampling U from Uniform(0,1) and computing:

$$
g = -\log(-\log(U))
$$

The soft weights are:

$$
w_i = \frac{\exp((s_i + g_i) / \tau)}{\sum_j \exp((s_j + g_j) / \tau)}
$$

where s are the scores, g are Gumbel noise samples, and tau is the temperature.

Temperature is annealed linearly from 1.0 to 0.1 over training, shifting from exploration to exploitation.

**Straight-Through Estimator (faster).** The forward pass uses hard argmax (fully discrete). The backward pass pretends argmax was the identity function. This is mathematically incorrect but works in practice.

**REINFORCE (fallback).** Treats program selection as a policy decision. Samples programs proportional to scores, uses the verifier reward as a signal, subtracts a baseline to reduce variance. No differentiability needed.

---

## Temperature annealing

$$
\tau(\text{step}) = \tau_{\text{start}} + \frac{\text{step}}{\text{anneal\_steps}} \cdot (\tau_{\text{end}} - \tau_{\text{start}})
$$

At step 0: tau = 1.0 (broad exploration)
At step 10 000: tau = 0.1 (sharp exploitation)

The gradient signal in the early training phase points in many directions simultaneously, which helps the encoder learn a diverse latent space. Late in training, the sharp temperature commits to the best programs.

---
---

# agent — the full loop

## What it does in plain terms

The agent runs the complete episode. It coordinates encoder, search, verifier, and bridge into a loop that either finds a verified rule or exhausts its budget.

The key design decision is info-gain exploration: when no rule is verified yet, the agent does not act randomly. It chooses the action that would most update its beliefs about which rule is correct.

---

## Episode loop

```mermaid
flowchart TD
    A[reset environment] --> B[encode obs history]
    B --> C[search for candidates]
    C --> D[verify best candidate]
    D --> E{verified?}
    E -->|yes| F[act greedily from rule]
    E -->|no| G[info-gain action]
    G --> H[observe transition]
    H --> I{budget exhausted?}
    I -->|no| B
    I -->|yes| J[return failed result]
    F --> K[return solved result]

    style F fill:#E1F5EE,stroke:#0F6E56,color:#085041
    style J fill:#FAECE7,stroke:#993C1D,color:#712B13
    style K fill:#E1F5EE,stroke:#0F6E56,color:#085041
```

---

## Info-gain action selection

When the verifier has not found a verified rule, the agent must explore. But random exploration is wasteful — it may reveal nothing useful about which rule is correct.

Info-gain exploration asks: "which action would most reduce my uncertainty about the rule?"

For each available action a, the agent simulates what each candidate program would predict as the outcome. Actions where the candidates disagree are more informative than actions where they all agree.

$$
\text{info\_gain}(a) \approx \text{fraction of cell positions where candidates disagree on outcome}
$$

The action with the highest info-gain is taken. After observing the real outcome, the candidates that predicted incorrectly can be eliminated.

This is active learning applied to rule induction: the agent designs its own experiments.

---

## Greedy acting from a verified rule

Once a rule is verified, exploration stops. The agent acts by choosing the action that the rule predicts will produce the most change.

This is a heuristic: ARC tasks typically require the agent to do something to the grid, so the action that causes the most transformation is usually correct. A more principled version would have the goal state encoded in the environment.

---

## The three metrics

The review identified three metrics as most informative:

**Success rate.** Did the agent solve the episode? This is the headline number but not the most useful for research.

**Mean exploration steps.** How many steps before the rule was verified? This directly maps to the ARC-AGI-3 efficiency score. The target is to stay well below the baseline action counts in the game data.

**Posterior concentration rate.** Does the bridge signal improve over the course of an episode? A positive slope means the encoder is learning to concentrate probability mass on better programs as evidence accumulates. This is the scientific signal that distinguishes "the system got lucky" from "the system is actually learning to reason."

$$
\text{posterior concentration} = \text{mean slope of bridge signal across episode steps}
$$

A system that truly forms hypotheses should show increasing bridge signals as evidence narrows the posterior. A system that is just pattern-matching should show flat or noisy signals.
# osmosis — architecture overview

osmosis is a research system that tries to solve interactive puzzles by forming compact rules, not by memorising patterns. It watches a few interactions, guesses what rule governs the environment, checks the guess, and acts efficiently from it.

This document explains how the system works in plain terms with the maths kept as simple as possible.

---

## The central idea

Most AI systems learn by memorising: "I have seen something like this before, here is what I did." osmosis learns by explaining: "here is the rule that generates what I see."

The difference matters because ARC-AGI-3 scores agents on how few steps they take to solve a level. A memoriser needs to see many examples before it is confident. An explainer can form the right hypothesis after two or three observations and then act directly from it.

The system is built around one question:

> Can we learn a continuous distribution over possible rules whose geometry respects what rules actually do?

If nearby points in the distribution correspond to rules with similar behaviour, then a search guided by that distribution will find the right rule quickly.

---

## The three cruxes

The architecture was designed around three hard problems that must all be solved for the system to work.

**Crux 1 — DSL bias.** The space of possible rules must be the right shape. If the language for expressing rules is too narrow, the correct rule cannot be expressed. If it is too broad, search is hopeless. The coarse-to-fine grammar in `dsl.py` is the answer: express rough sketches first, refine promising ones second.

**Crux 2 — Proposal calibration.** The continuous distribution must line up with what rules actually do. Two rules that behave identically on all inputs should be neighbours in the distribution. Two rules that behave differently should be far apart. The contrastive loss in `proposer.py` enforces this.

**Crux 3 — Bridge stability.** Gradients must flow from the discrete verifier back to the continuous encoder. The verifier says "this rule is 0.8 correct." The encoder must be able to use that signal to improve. The bridge in `bridge.py` provides three ways to do this, with different tradeoffs.

---

## Module map

```mermaid
graph TD
    A[environment] -->|state, action, next_state| B[encoder]
    B -->|latent z| C[proposer]
    C -->|candidate programs| D[search]
    D -->|top candidates| E[verifier]
    E -->|soft scores| F[bridge]
    F -->|gradient signal| B
    F -->|selected program| G[agent]
    G -->|info-gain action| A

    style B fill:#EEEDFE,stroke:#534AB7,color:#3C3489
    style C fill:#FAEEDA,stroke:#854F0B,color:#633806
    style D fill:#E1F5EE,stroke:#0F6E56,color:#085041
    style E fill:#E1F5EE,stroke:#0F6E56,color:#085041
    style F fill:#FAECE7,stroke:#993C1D,color:#712B13
    style G fill:#F1EFE8,stroke:#5F5E5A,color:#444441
```

---

## Data flow in one episode

```mermaid
sequenceDiagram
    participant Env as environment
    participant Enc as encoder
    participant Src as search
    participant Ver as verifier
    participant Br as bridge
    participant Agt as agent

    Env->>Agt: reset, initial state
    loop until verified or budget exhausted
        Agt->>Enc: observation history
        Enc->>Src: latent z
        Src->>Ver: top-m candidate programs
        Ver->>Br: soft scores
        Br->>Agt: selected program + gradient signal
        alt rule verified
            Agt->>Env: greedy action from rule
        else not verified
            Agt->>Env: info-gain action
            Env->>Agt: new transition
        end
    end
```

---

## Scoring

ARC-AGI-3 gives a lower score the more steps an agent takes. The efficiency metric used here is:

$$
\text{efficiency} = \frac{1}{\text{steps to verified rule}}
$$

Solving in 1 step gives efficiency 1.0. Solving in 10 steps gives 0.1. Exhausting the budget without solving gives 0.0.

The three metrics tracked by `OsmosisMetrics`:

| Metric | What it measures |
|---|---|
| success rate | fraction of episodes solved |
| mean steps | average steps to verified rule |
| posterior concentration | does the bridge signal improve over the episode? |

The third metric is the most informative for research: a positive slope in bridge signals means the encoder is learning to concentrate probability mass on better programs as evidence accumulates.

---

## File index

| File | Role | Crux |
|---|---|---|
| `encoder.py` | observation sequence to latent distribution | — |
| `dsl.py` | coarse-to-fine rule language | 1 |
| `proposer.py` | contrastive alignment of latent geometry | 2 |
| `search.py` | GPU-parallel candidate search | — |
| `verifier.py` | soft four-signal verification | 3 support |
| `bridge.py` | gradient flow: discrete to continuous | 3 |
| `agent.py` | full episode loop and metrics | — |
"""
osmosis/proposer.py

Behavioral contrastive alignment of the encoder's latent space.

The core problem this solves (Crux 2):
  The encoder produces a smooth latent z. But "smooth" is not the same
  as "useful for program search." Two latents can be close in z-space
  while corresponding to completely different rules — which means the
  GPU search wastes compute on irrelevant candidates.

  This module forces the geometry to respect behavioral equivalence:
  programs that produce the same transitions should be neighbors in z.
  Programs that behave differently should be separated.

How it works:
  1. Sample pairs of observation sequences (from same or different rules)
  2. Encode both → (z_i, z_j)
  3. Execute candidate programs on probe states → behavior vectors
  4. If behavior(i) ≈ behavior(j): pull z_i and z_j together
     If behavior(i) ≠ behavior(j): push them apart (up to margin)

This is a contrastive loss — the same idea used in SimCLR / CLIP, but
applied to program semantics rather than image augmentations.
"""

from __future__ import annotations

import numpy as np
from dataclasses import dataclass
from typing import Sequence

from osmosis.dsl import DSL, GridState, Action, Program
from osmosis.encoder import ObsEncoder, Transition, LatentDist



# Probe — a (state, action) pair used to test program behavior


@dataclass
class Probe:
    """
    A held-out (state, action) pair used to measure what a program does.
    Probes are not part of the agent's observation history —
    they are synthetic test inputs used only during training.
    """
    state:  GridState
    action: Action


def random_probe(
    grid_h: int,
    grid_w: int,
    rng: np.random.Generator,
) -> Probe:
    grid = rng.integers(0, 10, (grid_h, grid_w)).astype(np.int8)
    action = Action(
        kind  = "keyboard",
        key   = rng.choice(["ArrowUp", "ArrowDown", "ArrowLeft", "ArrowRight"]),
        row   = int(rng.integers(0, grid_h)),
        col   = int(rng.integers(0, grid_w)),
    )
    return Probe(state=GridState(grid), action=action)



# Behavior vector — fingerprint of a program's input/output behavior


def behavior_vector(
    program: Program,
    probes:  list[Probe],
    dsl:     DSL,
) -> np.ndarray:
    """
    Execute program on each probe → flatten outputs → behavior vector.

    Two programs are behaviorally equivalent iff their behavior vectors
    are identical (or very close, for noisy environments).

    Shape: (n_probes * grid_h * grid_w,)
    """
    parts = []
    for probe in probes:
        out = dsl.execute(program, probe.state, probe.action)
        parts.append(out.flatten().astype(np.float32))
    return np.concatenate(parts)


def behavioral_similarity(
    bvec_i: np.ndarray,
    bvec_j: np.ndarray,
    threshold: float = 0.95,
) -> bool:
    """
    Two programs are 'same behavior' if their output vectors agree
    on >= threshold fraction of cells.
    """
    agree = np.mean(bvec_i == bvec_j)
    return bool(agree >= threshold)



# Contrastive loss


def contrastive_loss(
    z_i:           np.ndarray,
    z_j:           np.ndarray,
    same_behavior: bool,
    margin:        float = 2.0,
) -> float:
    """
    Standard contrastive loss (LeCun et al., 2005).

    same_behavior = True  (positive pair):
        loss = ||z_i - z_j||^2
        Pull the two latents together.

    same_behavior = False (negative pair):
        loss = max(0, margin - ||z_i - z_j||)^2
        Push them apart — but only if they are closer than `margin`.
        Once they are far enough, loss = 0 (no gradient).

    Parameters
    ----------
    z_i, z_j      : latent vectors, shape (d_latent,)
    same_behavior : whether the two programs behave identically on probes
    margin        : minimum desired separation for negative pairs
    """
    dist = float(np.linalg.norm(z_i - z_j))

    if same_behavior:
        return dist ** 2
    else:
        return max(0.0, margin - dist) ** 2


def batch_contrastive_loss(
    latents:   list[np.ndarray],
    behaviors: list[np.ndarray],
    margin:    float = 2.0,
    max_pairs: int   = 256,
    rng:       np.random.Generator | None = None,
) -> float:
    """
    Compute mean contrastive loss over a random subset of pairs.

    Parameters
    ----------
    latents   : list of z vectors, one per program
    behaviors : list of behavior vectors, one per program
    margin    : separation margin for negative pairs
    max_pairs : cap on number of pairs (avoids O(n^2) blowup)
    rng       : random generator for pair sampling
    """
    rng = rng or np.random.default_rng()
    n   = len(latents)
    if n < 2:
        return 0.0

    total, count = 0.0, 0
    pairs = [(i, j) for i in range(n) for j in range(i+1, n)]
    if len(pairs) > max_pairs:
        idx   = rng.choice(len(pairs), size=max_pairs, replace=False)
        pairs = [pairs[k] for k in idx]

    for i, j in pairs:
        same = behavioral_similarity(behaviors[i], behaviors[j])
        total += contrastive_loss(latents[i], latents[j], same, margin)
        count += 1

    return total / count if count > 0 else 0.0



# Proposer


class Proposer:
    """
    Maps a latent z to a ranked list of candidate programs.

    In the numpy reference implementation, ranking uses cosine similarity
    between z and precomputed program prototype embeddings.

    In the torch training version, this becomes a learned MLP that
    directly outputs program scores — trained end-to-end with the
    contrastive loss.

    Usage
    -----
    proposer = Proposer(encoder, dsl, n_probes=16)
    proposer.build_index(n_programs=512)          # precompute prototypes
    programs, scores = proposer.propose(z, n=32)  # rank candidates
    loss = proposer.alignment_loss(z_i, z_j, prog_i, prog_j)
    """

    def __init__(
        self,
        encoder:   ObsEncoder,
        dsl:       DSL,
        n_probes:  int = 16,
        rng_seed:  int = 42,
    ):
        self.encoder  = encoder
        self.dsl      = dsl
        self.n_probes = n_probes
        self.rng      = np.random.default_rng(rng_seed)

        # Probe set — fixed during training for consistent comparison
        self.probes: list[Probe] = [
            random_probe(dsl.grid_h, dsl.grid_w, self.rng)
            for _ in range(n_probes)
        ]

        # Program index (built lazily)
        self._programs:  list[Program]    = []
        self._behaviors: list[np.ndarray] = []
        self._prototypes: np.ndarray | None = None  # (n_prog, d_latent)

    # ------------------------------------------------------------------
    # Index construction
    # ------------------------------------------------------------------

    def build_index(self, n_programs: int = 512) -> None:
        """
        Enumerate programs from the DSL, compute their behavior vectors,
        and build prototype latents for similarity-based ranking.

        The prototype for program p is the mean latent across synthetic
        trajectories that are consistent with p's behavior. In the numpy
        reference, we approximate this with random latent vectors shaped
        by behavioral clustering.
        """
        sketches = self.dsl.sketch(n=n_programs // 8)
        programs: list[Program] = []
        for sketch in sketches[:n_programs // 8]:
            programs.extend(self.dsl.refine(sketch, n=8))
            if len(programs) >= n_programs:
                break
        programs = programs[:n_programs]

        behaviors = [
            behavior_vector(p, self.probes, self.dsl) for p in programs
        ]

        # Prototype = random unit vector in latent space, grouped by
        # behavioral equivalence class. Programs with same behavior share
        # a prototype direction (with small noise).
        d = self.encoder.d_latent
        equiv_map: dict[bytes, np.ndarray] = {}
        prototypes = np.zeros((len(programs), d), dtype=np.float32)

        for k, bvec in enumerate(behaviors):
            key = bvec.tobytes()
            if key not in equiv_map:
                v = self.rng.standard_normal(d).astype(np.float32)
                equiv_map[key] = v / (np.linalg.norm(v) + 1e-8)
            noise = self.rng.standard_normal(d).astype(np.float32) * 0.05
            prototypes[k] = equiv_map[key] + noise

        self._programs   = programs
        self._behaviors  = behaviors
        self._prototypes = prototypes

    # ------------------------------------------------------------------
    # Proposal
    # ------------------------------------------------------------------

    def propose(
        self,
        z: np.ndarray,
        n: int = 32,
    ) -> tuple[list[Program], np.ndarray]:
        """
        Rank candidate programs by cosine similarity between z and
        their prototype embeddings.

        Returns
        -------
        programs : top-n programs
        scores   : cosine similarity scores, shape (n,)
        """
        if self._prototypes is None or len(self._programs) == 0:
            self.build_index()

        z_norm  = z / (np.linalg.norm(z) + 1e-8)
        p_norm  = self._prototypes / (
            np.linalg.norm(self._prototypes, axis=1, keepdims=True) + 1e-8
        )
        sims    = p_norm @ z_norm                   # (n_prog,)
        top_idx = np.argsort(sims)[::-1][:n]

        programs = [self._programs[i] for i in top_idx]
        scores   = sims[top_idx]
        return programs, scores

    # ------------------------------------------------------------------
    # Alignment loss (used in training loop)
    # ------------------------------------------------------------------

    def alignment_loss(
        self,
        z_i:     np.ndarray,
        z_j:     np.ndarray,
        prog_i:  Program,
        prog_j:  Program,
        margin:  float = 2.0,
    ) -> float:
        """
        Compute contrastive loss between two latents given their programs.
        Behavioral equivalence is determined by executing both programs
        on the fixed probe set.
        """
        bvec_i = behavior_vector(prog_i, self.probes, self.dsl)
        bvec_j = behavior_vector(prog_j, self.probes, self.dsl)
        same   = behavioral_similarity(bvec_i, bvec_j)
        return contrastive_loss(z_i, z_j, same, margin)

    def batch_alignment_loss(
        self,
        latents:  list[np.ndarray],
        programs: list[Program],
        margin:   float = 2.0,
    ) -> float:
        """Compute mean alignment loss over a batch of (z, program) pairs."""
        behaviors = [behavior_vector(p, self.probes, self.dsl) for p in programs]
        return batch_contrastive_loss(latents, behaviors, margin, rng=self.rng)
"""
osmosis/search.py

GPU-parallel program search guided by the encoder's latent z.

The core idea (from the blueprint):
  Instead of searching programs one at a time, execute k candidates
  simultaneously. On GPU this is a single kernel call.
  The proposer biases sampling toward the latent's high-probability
  region so we're not searching blindly.

Two-phase coarse-to-fine strategy:
  Phase 1 — Coarse (Level 0 sketches):
    Sample 4 096 sketches, execute on first 4 observations only.
    Prune to top-128 by consistency score. Fast.

  Phase 2 — Fine (Level 1 refinements):
    Expand top-128 sketches × 512 refinements = 65 536 candidates.
    Execute on full observation sequence. Score by soft verifier.

This file contains the CPU reference implementation.
The real GPU version replaces the inner loops with a CUDA kernel that
vectorises the DSL interpreter across the k-batch dimension.
See comments marked [GPU] for where the kernel calls go.
"""

from __future__ import annotations

import numpy as np
from dataclasses import dataclass
from typing import Sequence

from osmosis.dsl import DSL, GridState, Action, Program
from osmosis.encoder import Transition



# Search result


@dataclass
class SearchResult:
    programs:  list[Program]    # top-m candidates after search
    scores:    np.ndarray       # consistency scores, shape (m,)
    n_executed: int             # total programs executed (throughput metric)

    @property
    def best(self) -> Program:
        return self.programs[0]

    @property
    def best_score(self) -> float:
        return float(self.scores[0])



# Consistency scoring


def consistency_score(
    program:  Program,
    obs_seq:  list[Transition],
    dsl:      DSL,
) -> float:
    """
    Fraction of observed transitions the program correctly predicts.

    score = (# transitions where program(s,a) == s') / len(obs_seq)

    This is the fast inner-loop score used during coarse search.
    The verifier (osmosis/verifier.py) adds richer partial-credit signals
    on top of this for the final ranking.
    """
    if not obs_seq:
        return 0.0
    correct = 0
    for t in obs_seq:
        state  = GridState.from_flat(t.state,  dsl.grid_h, dsl.grid_w)
        action = _vec_to_action(t.action)
        s_next = GridState.from_flat(t.next_state, dsl.grid_h, dsl.grid_w)
        pred   = dsl.execute(program, state, action)
        if pred == s_next:
            correct += 1
    return correct / len(obs_seq)


def _vec_to_action(vec: np.ndarray) -> Action:
    """Recover a minimal Action from an encoded action vector."""
    return Action(kind="keyboard", key="ArrowRight")



# Batch consistency (CPU reference; replaced by CUDA kernel in GPU version)


def batch_consistency(
    programs: list[Program],
    obs_seq:  list[Transition],
    dsl:      DSL,
    subsample: int | None = None,
) -> np.ndarray:
    """
    Compute consistency scores for k programs over obs_seq.

    Returns shape (k,) float32.

    [GPU] This loop is the vectorisation target.
    Replace with:
        scores = cuda_exec(programs_tensor, obs_tensor)
    where cuda_exec runs all k programs in a single kernel dispatch.

    subsample: if set, only evaluate on this many transitions (faster coarse pass).
    """
    seq = obs_seq[:subsample] if subsample else obs_seq
    scores = np.zeros(len(programs), dtype=np.float32)
    for i, prog in enumerate(programs):
        scores[i] = consistency_score(prog, seq, dsl)
    return scores



# Main search function


def search(
    z:        np.ndarray,
    obs_seq:  list[Transition],
    dsl:      DSL,
    proposer: object | None = None,
    k_coarse: int = 256,
    k_refine: int = 16,
    n_refine: int = 8,
    top_m:    int = 8,
    rng:      np.random.Generator | None = None,
) -> SearchResult:
    """
    Two-phase coarse-to-fine program search.

    Parameters
    ----------
    z         : latent from encoder, used to bias program sampling
    obs_seq   : interaction history so far
    dsl       : DSL instance
    proposer  : optional Proposer for z-guided candidate ranking
    k_coarse  : number of Level 0 sketches to evaluate in Phase 1
    k_refine  : number of top sketches to refine in Phase 2
    n_refine  : refinements per sketch
    top_m     : number of final candidates to return
    rng       : random generator

    Returns
    -------
    SearchResult with top-m programs and their scores
    """
    rng = rng or np.random.default_rng()
    n_executed = 0

    # ------------------------------------------------------------------
    # Phase 1 — Coarse sketch search
    # ------------------------------------------------------------------

    if proposer is not None:
        # Use proposer to bias toward high-probability sketches
        sketches, _ = proposer.propose(z, n=k_coarse)
    else:
        sketches = dsl.sketch(z=z, n=k_coarse)

    # [GPU] Replace loop with cuda_exec on (k_coarse, H, W) batch
    # Subsample: use only first 4 obs for speed in coarse pass
    subsample = min(4, len(obs_seq)) if len(obs_seq) > 0 else None
    coarse_scores = batch_consistency(sketches, obs_seq, dsl, subsample=subsample)
    n_executed += len(sketches)

    # Keep top-k_refine sketches for refinement
    top_sketch_idx = np.argsort(coarse_scores)[::-1][:k_refine]
    top_sketches   = [sketches[i] for i in top_sketch_idx]

    # ------------------------------------------------------------------
    # Phase 2 — Fine refinement search
    # ------------------------------------------------------------------

    refined_programs: list[Program] = []
    for sketch in top_sketches:
        refined_programs.extend(dsl.refine(sketch, z=z, n=n_refine))

    if not refined_programs:
        # Fallback: return top sketches directly
        top_sketches_sorted = [sketches[i] for i in top_sketch_idx[:top_m]]
        scores = coarse_scores[top_sketch_idx[:top_m]]
        return SearchResult(top_sketches_sorted, scores, n_executed)

    # [GPU] Replace loop with cuda_exec on full obs_seq
    fine_scores = batch_consistency(refined_programs, obs_seq, dsl)
    n_executed += len(refined_programs)

    # Select top-m final candidates
    top_fine_idx  = np.argsort(fine_scores)[::-1][:top_m]
    top_programs  = [refined_programs[i] for i in top_fine_idx]
    top_scores    = fine_scores[top_fine_idx]

    return SearchResult(
        programs   = top_programs,
        scores     = top_scores,
        n_executed = n_executed,
    )



# Throughput benchmark (run this to measure Phase 2 speed)


def benchmark_throughput(
    dsl:      DSL,
    n_obs:    int   = 5,
    k_total:  int   = 512,
    rng_seed: int   = 0,
) -> dict:
    """
    Measure programs-per-second on this machine.
    Gives baseline for GPU speedup estimation.
    """
    import time
    rng = np.random.default_rng(rng_seed)

    obs_seq = []
    for _ in range(n_obs):
        s  = rng.integers(0, 10, (dsl.grid_h, dsl.grid_w)).astype(np.float32).flatten()
        a  = np.zeros(64, dtype=np.float32)
        s2 = rng.integers(0, 10, (dsl.grid_h, dsl.grid_w)).astype(np.float32).flatten()
        obs_seq.append(Transition(state=s, action=a, next_state=s2))

    programs = dsl.sketch(n=k_total)

    t0     = time.perf_counter()
    scores = batch_consistency(programs, obs_seq, dsl)
    dt     = time.perf_counter() - t0

    return {
        "n_programs":   k_total,
        "n_obs":        n_obs,
        "time_s":       round(dt, 3),
        "programs_per_s": round(k_total / dt),
        "note": "CPU reference. GPU target: 100-1000x faster."
    }

class MockEnv:
    """Minimal mock ARC environment for agent testing."""
    def __init__(self, h=5, w=5, rng_seed=0):
        self.h   = h
        self.w   = w
        self.rng = np.random.default_rng(rng_seed)
        self._state   = None
        self._solved  = False
        self._step_n  = 0

    def reset(self) -> GridState:
        self._state  = GridState(self.rng.integers(0,10,(self.h,self.w)).astype(np.int8))
        self._solved = False
        self._step_n = 0
        return self._state

    def step(self, action: Action):
        self._step_n += 1
        new_grid = self.rng.integers(0, 10, (self.h, self.w)).astype(np.int8)
        self._state = GridState(new_grid)
        # Randomly solve after 3+ steps (for testing)
        self._solved = self._step_n >= 3
        return self._state, self._solved

    def current_state(self) -> GridState:
        return self._state

    def action_space(self) -> list[Action]:
        return [
            Action(kind="keyboard", key="ArrowUp"),
            Action(kind="keyboard", key="ArrowDown"),
            Action(kind="keyboard", key="ArrowLeft"),
            Action(kind="keyboard", key="ArrowRight"),
        ]

    def is_solved(self) -> bool:
        return self._solved

"""
osmosis/verifier.py

Soft symbolic verification with four partial-credit signals.

Why soft verification matters:
  Hard pass/fail (1 if program explains all transitions, 0 otherwise)
  makes training miserable. Most programs are wrong, so most training
  steps produce zero gradient — the bridge cannot learn.

  Soft verification gives the encoder a gradient signal even when no
  program is fully correct yet. It tells the encoder HOW wrong each
  candidate is, not just that it's wrong.

Four signals, each in [0, 1]:
  1. consistency     — fraction of observed transitions correctly predicted
  2. compression     — shorter programs are preferred (Occam's razor)
  3. action_pred     — does the program predict which actions matter?
  4. counterfactual  — does the rule generalise to held-out probe states?

Final reward = weighted sum. Full verification requires consistency == 1.0.
"""

from __future__ import annotations

import numpy as np
from dataclasses import dataclass

from osmosis.dsl import DSL, GridState, Action, Program
from osmosis.encoder import Transition
from osmosis.proposer import Probe, random_probe, behavior_vector



# Verification result


@dataclass
class VerifyResult:
    verified:       bool    # True only if consistency == 1.0
    reward:         float   # weighted soft score in [0, 1]
    consistency:    float   # fraction of transitions correctly predicted
    compression:    float   # program brevity score
    action_pred:    float   # action relevance score
    counterfactual: float   # generalisation to held-out probes

    def __repr__(self) -> str:
        v = "VERIFIED" if self.verified else "partial"
        return (f"VerifyResult({v} | reward={self.reward:.3f} | "
                f"cons={self.consistency:.2f} comp={self.compression:.2f} "
                f"act={self.action_pred:.2f} cf={self.counterfactual:.2f})")



# Individual scoring functions


def score_consistency(
    program: Program,
    obs_seq: list[Transition],
    dsl:     DSL,
) -> float:
    """
    Fraction of observed (s, a, s') triples that the program predicts correctly.

    score = correct_predictions / len(obs_seq)

    This is the primary signal. A program must score 1.0 here to be
    considered fully verified.
    """
    if not obs_seq:
        return 0.0
    correct = 0
    for t in obs_seq:
        state  = GridState.from_flat(t.state,      dsl.grid_h, dsl.grid_w)
        s_next = GridState.from_flat(t.next_state, dsl.grid_h, dsl.grid_w)
        action = Action(kind="keyboard", key="ArrowRight")
        pred   = dsl.execute(program, state, action)
        if pred == s_next:
            correct += 1
    return correct / len(obs_seq)


def score_compression(program: Program, max_len: int = 6) -> float:
    """
    Prefer shorter programs (Occam's razor).
    A program of length 1 scores 1.0; length max_len scores ~0.17.

    score = 1 / program_length

    Shorter programs are more likely to be the true underlying rule —
    ARC tasks are designed to have compact solutions.
    """
    length = max(1, len(program))
    return 1.0 / length


def score_action_prediction(
    program:  Program,
    obs_seq:  list[Transition],
    dsl:      DSL,
) -> float:
    """
    Does applying the program ever produce a non-trivial change?

    A program that always returns the input state (degenerate identity)
    should score 0. A program that produces meaningful changes scores higher.

    Measured as: fraction of transitions where program output != input state.
    """
    if not obs_seq:
        return 0.0
    nontrivial = 0
    for t in obs_seq:
        state  = GridState.from_flat(t.state, dsl.grid_h, dsl.grid_w)
        action = Action(kind="keyboard", key="ArrowRight")
        pred   = dsl.execute(program, state, action)
        if pred != state:
            nontrivial += 1
    return nontrivial / len(obs_seq)


def score_counterfactual(
    program: Program,
    probes:  list[Probe],
    dsl:     DSL,
    rng:     np.random.Generator | None = None,
) -> float:
    """
    Does the program behave consistently on held-out probe states?

    We measure internal consistency: if a program's behavior is stable
    and deterministic across diverse inputs, it's more likely to encode
    a genuine rule rather than overfitting to observed transitions.

    Score = fraction of probe pairs where program behaves consistently
    (i.e., similar inputs produce similar outputs).
    """
    if not probes:
        return 0.5  # neutral if no probes

    outputs = []
    for probe in probes:
        out = dsl.execute(program, probe.state, probe.action)
        outputs.append(out.grid)

    # Measure output diversity — a good rule should be consistent
    # (not output a different random thing every time)
    if len(outputs) < 2:
        return 0.5

    diffs = []
    for i in range(len(outputs) - 1):
        diff = np.mean(outputs[i] != outputs[i+1])
        diffs.append(diff)

    # High diversity = inconsistent rule = lower score
    mean_diff = float(np.mean(diffs))
    return float(np.clip(1.0 - mean_diff, 0.0, 1.0))



# Weights


DEFAULT_WEIGHTS = {
    "consistency":    0.60,
    "compression":    0.15,
    "action_pred":    0.10,
    "counterfactual": 0.15,
}



# Verifier


class Verifier:
    """
    Soft symbolic verifier. Scores candidate programs on four dimensions
    and returns a VerifyResult with a scalar reward for the bridge.

    Usage
    -----
    verifier = Verifier(dsl, n_probes=16)
    result = verifier.verify(program, obs_seq)
    results = verifier.rank(programs, obs_seq)  # rank a list
    """

    def __init__(
        self,
        dsl:       DSL,
        n_probes:  int   = 16,
        weights:   dict  | None = None,
        rng_seed:  int   = 42,
    ):
        self.dsl     = dsl
        self.weights = weights or DEFAULT_WEIGHTS.copy()
        self.rng     = np.random.default_rng(rng_seed)
        self.probes: list[Probe] = [
            random_probe(dsl.grid_h, dsl.grid_w, self.rng)
            for _ in range(n_probes)
        ]

    def verify(
        self,
        program: Program,
        obs_seq: list[Transition],
    ) -> VerifyResult:
        """
        Score a single program on all four dimensions.

        Returns VerifyResult. verified=True only when consistency==1.0.
        """
        w = self.weights

        cons  = score_consistency(program, obs_seq, self.dsl)
        comp  = score_compression(program)
        act   = score_action_prediction(program, obs_seq, self.dsl)
        cf    = score_counterfactual(program, self.probes, self.dsl, self.rng)

        reward = (
            w["consistency"]    * cons +
            w["compression"]    * comp +
            w["action_pred"]    * act  +
            w["counterfactual"] * cf
        )

        return VerifyResult(
            verified       = (cons >= 1.0 and len(obs_seq) > 0),
            reward         = float(np.clip(reward, 0.0, 1.0)),
            consistency    = cons,
            compression    = comp,
            action_pred    = act,
            counterfactual = cf,
        )

    def rank(
        self,
        programs: list[Program],
        obs_seq:  list[Transition],
    ) -> list[tuple[Program, VerifyResult]]:
        """
        Verify all programs and return sorted by reward (best first).
        """
        results = [(p, self.verify(p, obs_seq)) for p in programs]
        return sorted(results, key=lambda x: x[1].reward, reverse=True)

    def best(
        self,
        programs: list[Program],
        obs_seq:  list[Transition],
    ) -> tuple[Program, VerifyResult]:
        """Return the single best program and its result."""
        ranked = self.rank(programs, obs_seq)
        return ranked[0]
